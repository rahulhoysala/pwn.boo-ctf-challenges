        }

`reg45gmiso48hf.txt`



The static analysis portion is more or less complete. When the player tries to run the binary, nothing visibly happens. This is because the program checks the current directory for the existence of a file called `reg45gmiso48hf.txt`. If it does not exist, the program exits. If it does, it calculates the SHA-256 hash of the txt file and POST's it to the server, along with the token. 
The token is accepted by the server and stored in its database with a TTL of 1 hour. 

This entire challenge is meant to simulate ransomware. I was inspired by the Wannacry ransomware which gained popularity in 2017. More particularly, I wanted to simulate evasion tactics employed by malware. This one was more directly inspired by Marcus Hutchins, who registered a domain that Wannacry was checking for as a 'kill switch'. Similarly, this binary checks for the existence of a file with a seemingly nonsensical name on the player's workstation before revealing its behavior, forcing the player to perform static analysis. If the file exists and if its content is the same as the file name itself, it calculates the SHA-256 hash and compares it with the server's pre-stored value (this was an alpha version - the hash computation was later changed to reflect the MD5 hash of "Halloween" since thats easier to reverse upon inspecting outgoing requests). You then visit the server manually and enter the "decryption key" provided by the binary, which then provides you with the flag.

The main handler in the server-side Cloudflare worker:

```typescript
export default {
  async fetch(request: Request, env: Env) {
    const url = new URL(request.url);
    const path = url.pathname;

    // Ensure DB table exists before handling mutating operations or status queries
    // (D1 is safe to call repeatedly). See D1 docs for usage from Workers. :contentReference[oaicite:6]{index=6}
    try {
      await ensureTable(env);
    } catch (err) {
      return textResponse(`<h1>DB error</h1><pre>${String(err)}</pre>`, 500);
    }

    // GET /ransomware-panel
    if (request.method === "GET" && path === "/ransomware-panel") {
      const dbExists = true; // D1 binding exists if configured
      const tokenCount = await dbCountTokens(env);
      return textResponse(INDEX_HTML(dbExists, tokenCount));
    }

    // POST /ransomware-panel/register  -> expects JSON { token, proof }
    if (request.method === "POST" && path === "/ransomware-panel/register") {
      let data: any = {};
      try {
        data = await request.json();
      } catch (e) {
        return jsonResponse({ error: "bad json" }, 400);
      }
      const token = data?.token;
      const proof = data?.proof;
      if (!token || !proof) {
        return jsonResponse({ error: "missing token or proof" }, 400);
      }
      // Validate proof exactly same as original
      if (proof !== EXPECTED_SHA256_HEX) {
        return jsonResponse({ error: "invalid proof" }, 403);
      }

      // Attempt insert
      try {
        await dbInsertToken(env, token, env.FLAG, TOKEN_TTL_SECONDS);
      } catch (err: any) {
        // SQLite unique constraint on token -> token exists
        if (
          err?.message &&
          (err.message.includes("UNIQUE") ||
            err.message.includes("unique") ||
            err.message.includes("constraint"))
        ) {
          return jsonResponse({ error: "token exists" }, 403);
        }
        return jsonResponse({ error: "db error", detail: String(err) }, 500);
      }
      // Success
      return jsonResponse({ status: "registered" }, 200);
    }

    // POST /ransomware-panel/redeem  -> Accept JSON or form
    if (request.method === "POST" && path === "/ransomware-panel/redeem") {
      // Try JSON first
      let token: string | undefined;
      const ctype = request.headers.get("content-type") ?? "";
      if (ctype.includes("application/json")) {
        try {
          const j: any = await request.json();
          token = j?.token;
        } catch {
          // ignore
        }
      } else {
        // form post from the UI
        try {
          const form = await request.formData();
          token = form.get("token")?.toString();
        } catch {
          // ignore
        }
      }

      if (!token) {
        // Redirect to /ransomware-panel with a flash-like query parameter (UI reads it)
        return Response.redirect(
          new URL(
            "/ransomware-panel?flash=" + encodeURIComponent("Missing token value"),
            request.url
          ),
          303
        );
      }

      const row = await dbGetTokenRow(env, token);
      if (!row) {
        return Response.redirect(
          new URL(
            "/ransomware-panel?flash=" + encodeURIComponent("Invalid token"),
            request.url
          ),
          303
        );
      }
      const { flag, created, ttl, used } = row;
      const now = Math.floor(Date.now() / 1000);
      if (used) {
        return Response.redirect(
          new URL(
            "/ransomware-panel?flash=" + encodeURIComponent("Token already used"),
            request.url
          ),
          303
        );
      }
      if (now > created + ttl) {
        return Response.redirect(
          new URL(
            "/ransomware-panel?flash=" + encodeURIComponent("Token expired"),
            request.url
          ),
          303
        );
      }

      await dbMarkUsed(env, token);
      // Render flag page
      return textResponse(FLAG_HTML(flag), 200);
    }

    // Fallback 404
    return new Response("Not Found", { status: 404 });
  },
};
```
